// api/prices.js (โค้ดสำหรับ Serverless Function บน Vercel)

const fetch = require('node-fetch');

// 1. กำหนด Item ID และ Locations ที่คุณต้องการ (T6-T8)
const itemIds = [
    "T6_BAG", 
    "T7_BAG", 
    "T8_BAG", 
    // คุณสามารถเพิ่ม Item ID อาวุธ ชุดเกราะ T6-T8 ทั้งหมดที่นี่
];

// แก้ไข Syntax Error: ใช้ [ ] แทน { }
const locations = [ 
    "Black Market", 
    "Bridgewatch", 
    "Martlock", 
    "Fort Sterling", 
    "Thetford", 
    "Lymhurst", 
    "Caerleon"
];

const QUALITY = 1; // Normal Quality

const FEE_RATE = 0.065; // 6.5% Black Market Fee

module.exports = async (req, res) => {
    try {
        // 2. สร้าง URL API
        const apiUrl = `https://west.albion-online-data.com/api/v2/stats/prices/${itemIds.join(',')}` +
                       `?locations=${locations.join(',')}` +
                       `&qualities=${QUALITY}`;

        // 3. ดึงข้อมูลจาก Albion API
        const response = await fetch(apiUrl);
        const data = await response.json();

        // 4. ประมวลผลและคำนวณกำไร
        
        // ก. หาข้อมูล Black Market (BM) Sell Price Max (O)
        const bmData = data.filter(d => d.city === "Black Market");
        const bmMap = new Map();
        bmData.forEach(item => {
            // buy_price_max ของ BM คือราคาที่ BM ซื้อจากคุณ
            const sellPriceGross = item.buy_price_max / (1 - FEE_RATE);
            bmMap.set(item.item_id, sellPriceGross);
        });

        // ข. คำนวณกำไรสำหรับเมืองหลัก
        const results = data
            .filter(d => d.city !== "Black Market" && d.sell_price_min > 0) // กรองเหลือแต่เมืองหลักที่ราคามี Sell Order (M > 0)
            .map(cityItem => {
                const itemId = cityItem.item_id;
                const buyPrice = cityItem.sell_price_min; 
                const sOrderPriceGross = bmMap.get(itemId) || 0; 

                if (sOrderPriceGross === 0 || sOrderPriceGross <= buyPrice) {
                    return null;
                }

                const revenueAfterFee = sOrderPriceGross * (1 - FEE_RATE);
                const finalProfit = revenueAfterFee - buyPrice; 

                return {
                    itemId: itemId,
                    city: cityItem.city,
                    buyPrice: buyPrice,
                    sellPriceGross: sOrderPriceGross.toFixed(0),
                    finalProfit: finalProfit.toFixed(0),
                    margin: ((finalProfit / buyPrice) * 100).toFixed(2) + '%'
                };
            })
            .filter(Boolean) 
            .sort((a, b) => b.finalProfit - a.finalProfit); // เรียงกำไรมากไปน้อย

        // 5. ส่งผลลัพธ์กลับไปให้เบราว์เซอร์
        res.setHeader('Content-Type', 'application/json'); 
        res.status(200).send(JSON.stringify(results));
    } catch (error) {
        res.status(500).send(JSON.stringify({ error: error.message }));
    }
};
